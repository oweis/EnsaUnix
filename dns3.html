#!/bin/sh
apt-get update
apt-get install bind9
echo "auto lo">/etc/network/interfaces
echo "iface lo inet loopback">>/etc/network/interfaces
echo "auto eth0">>/etc/network/interfaces
echo "iface eth0 inet static">>/etc/network/interfaces
ifconfig>ifconfig.txt
answer=head -2 ifconfig.txt|grep inet|awk '{print $2}'|awk -F ":" '{print $2}'
echo "address $answer">>/etc/network/interfaces
echo "netmask 255.255.255.0">>/etc/network/interfaces
echo "dns-nameservers $answer">>/etc/network/interfaces
echo "donnez ton nom de domaine"
read dom
echo "zone "$dom"{">>/etc/bind/named.conf.local
echo "type master;">>/etc/bind/named.conf.local
echo "file "/etc/bind/db.$dom";">>/etc/bind/named.conf.local
echo "};">>/etc/bind/named.conf.local
cp db.empty db.$dom
echo "domain 192.168.8.101">/etc/bind/resolv.conf
echo "search enaunix.com">>/etc/bind/resolv.conf
head -6 /etc/bind/db.$dom>EnsaUnixunix.txt
echo "@ IN SOA db.$dom. root.localhost.(">>EnsaUnixunix.txt
tail -7 /etc/bind/db.$dom>>EnsaUnixunix.txt
echo "@ IN A $answer">>EnsaUnixunix.txt
rm /etc/bind/db.$dom
mv EnsaUnixunix.txt /etc/bind/db.$dom
service bind9 reload
