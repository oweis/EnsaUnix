WebDav



#!/bin/sh
n=`cat /etc/apache2/sites-available/000-default.conf|wc -l` m=$(($n-3)) mkdir /var/www/webdav
a2enmod dav
a2enmod dav_fs
nano fich
echo "DavLockDB /var/www/DavLock">fich
head -$m>>fich
echo "Alias /webdav /var/www/webdav">>fich
echo "">>fich
echo "DAV On">>fich
echo "">>fich
rm -f /etc/apache2/sites-available/000-default.conf
mv fich /etc/apache2/sites-available/000-default.conf
service apache2 restart
echo " si vous voulez partager un fichier"
echo "mettez une copie dans /var/www/webdav"
echo "et tapez dans ton terminal : service apache2 restart"
